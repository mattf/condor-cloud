    def self.convert_condor_state(state_id)
      case state_id
        when 0,1,5 then 'PENDING'
        when 2     then 'RUNNING'
        when 3,4   then 'SHUTTING_DOWN'
        else raise "Unknown Condor state (#{state_id})"
      end
    end
